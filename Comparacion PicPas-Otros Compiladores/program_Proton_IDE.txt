Device = 16F84A
Declare Xtal = 4

TRISA = $F0   ' RA0 to RA4 Outputs. 
INTCON.7 = 0  ' Global Interrupt off.

Inicio:        // bucle
  GoSub derecha
  GoSub izquierda
GoTo Inicio    // infinito

derecha: 
   PORTA.0 = 1
   DelayMS 300
   PORTA.0 = 0
   PORTA.1 = 1
   DelayMS 300
   PORTA.1 = 0
   PORTA.2 = 1
   DelayMS 300
   PORTA.2 = 0
   PORTA.3 = 1
   DelayMS 300
Return

izquierda:
   PORTA.3 = 0
   PORTA.2 = 1
   DelayMS 300
   PORTA.2 = 0
   PORTA.1 = 1
   DelayMS 300
   PORTA.1 = 0
Return
